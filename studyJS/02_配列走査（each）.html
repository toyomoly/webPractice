基本のforループ

var hoge = ["hoge", "fuga", "piyo"];
for (var i=0, len=hoge.length; i<len; i++) {
    console.log(hoge[i]);
}
  • 普通

Array.forEach

var hoge = ["hoge", "fuga", "piyo"];
hoge.forEach(function (ele, idx, ary) {
    console.log(ele);
});
  • Arrayのメソッド
  • 今時はコレを使うのが一番いい？

$.each

var hoge = ["hoge", "fuga", "piyo"];
$.each(hoge, function (idx, ele) {
    console.log(ele);
});
  • jQueryのメソッド
  • イテレータの引数がArray.forEachと異なるのでややこしい
  • thisも使えるが、オブジェクトに変換されているので少し注意
  • 配列っぽいもの（NodeListとか）にも使える

$(hoge).each

var hoge = ["hoge", "fuga", "piyo"];
$(hoge).each(function (idx, ele) {
    console.log(ele);
});
  • jQueryのメソッド
  • $.eachとほとんど一緒
