基本のforループ

var result = [];
var hoge = ["hoge", "fuga", "piyo"];
for (var i=0, len=hoge.length; i<len; i++) {
    result.push({name: hoge[i]});
}
  • 普通

Array.map

var hoge = ["hoge", "fuga", "piyo"];
var result = hoge.map(function (ele, idx, ary) {
    return {name: ele};
});
  • Arrayのメソッド
  • 今時はコレを使うのが一番いい？

$.map

var hoge = ["hoge", "fuga", "piyo"];
var result = $.map(hoge, function (ele, idx) {
    return {name: ele};
});
  • jQueryのメソッド
  • $.eachと結構違う、引数とか
  • thisは使えない
  • 配列っぽいもの（NodeListとか）にも使える

$(hoge).map

var hoge = ["hoge", "fuga", "piyo"];
var result = $(hoge).map(function (idx, ele) {
    return {name: ele};
});
  • jQueryのメソッド
  • $(hoge).eachと使い勝手は一緒
